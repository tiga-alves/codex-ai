# AENDERUNGEN:
#
# AN 2002.09.30
# Split Segment E2EDKA1 über Qualifier LF, AG, WE, AP, AB, RE, OMI, OSF, OPB, OSU, X (Rest)
#
# AN 2002.12.10
# - E2EDP16a hinzugefuegt, wird wenn benutzt, immer vor den Standard E2EDP16 Segmenten erscheinen.
#   Wird benoetigt, wenn Gruppen die auserhalb einer nonterminalen Zuordnung sind ebenfalls E2EDP16 Segmente
#   erzeugen sollen ( zB.: Sonderbehandlung fuer Rueckstandsmengen ).
# - Segment E2EDP16 ueber Gruppe GP16: Erzwingt die Abbildung der Variablen ins Segment E2EDP16 bei Schliessung
#   der Gruppe GP16 --> Nonterminale Abbildung auf GP16 !
#
# AN 2004.01.13
# - E2EDK11 * 9999   -->  GP11 * 9999  -  E2EDK11 +1
#   E2EDK11 Segment in Gruppe K11 eingeordnet. Segment darf dort nur einmal vorkommen, die Gruppe K11 kann
#   wiederholt werden. Damit ist das schliessen und Abbilden der Werte in die Felder des Segmentes beim
#   initialiseren eines neuen Segmentes gewaehrleistet.
#

FILE.

FILE    :
        SAPIDOC         [0:"EDI_DC40"]       + 9999
        ;

SAPIDOC :
        EDI_DC40        [0:"EDI_DC40"]       +    1
        G1              [0:"E2EDK09"]        +    1
        ;

G1      :
        E2EDK09         [0:"E2EDK09"]        +    1
        Z274K09         [0:"Z274K09"]        *    1
        E2EDK10         [0:"E2EDK10"]        *    1
        E2EDKA1x        [0:"E2EDKA1"]        +   10
        GP11            [0:"E2EDK11"]        * 9999
        GP11_DR         [0:"XXXXXXX"]        * 9999
        GP11_LF         [0:"XXXXXXX"]        * 9999
        G2              [0:"E2EDP10"]        + 9999
        ;


G2      :
        E2EDP10         [0:"E2EDP10"]        + 9999
        E2EDP11         [0:"E2EDP11"]        * 9999
        GP14            [0:"E2EDP14"]        * 9999 
        E2EDP15         [0:"E2EDP15"]        * 9999
        E2EDP16a        [0:"XXXXXXX"]        *   20
        GP16            [0:"E2EDP16"]        * 9999
        ;

GP16    :
        E2EDP16         [0:"E2EDP16"]        +    1
        ;

GP14    :
        E2EDP14         [0:"E2EDP14"]        +    1
        ;

        E2EDK11         [0:"E2EDK11"]        +    1
        ;

GP11_DR :
        E2EDK11_DR      [0:"E2EDK11"]        +    1
        ;

GP11_LF :
        E2EDK11_LF      [0:"E2EDK11"]        +    1
        ;

E2EDKA1x:
        E2EDKA1_LF      [? (KEY_EXISTS(0,"E2EDKA1") != 0 & KEY_EXISTS(63,"LF ")  != 0)]  |
        E2EDKA1_AG      [? (KEY_EXISTS(0,"E2EDKA1") != 0 & KEY_EXISTS(63,"AG ")  != 0)]  |
        E2EDKA1_AP      [? (KEY_EXISTS(0,"E2EDKA1") != 0 & KEY_EXISTS(63,"AP ")  != 0)]  |
        E2EDKA1_WE      [? (KEY_EXISTS(0,"E2EDKA1") != 0 & KEY_EXISTS(63,"WE ")  != 0)]  |
        E2EDKA1_RE      [? (KEY_EXISTS(0,"E2EDKA1") != 0 & KEY_EXISTS(63,"RE ")  != 0)]  |
        E2EDKA1_AB      [? (KEY_EXISTS(0,"E2EDKA1") != 0 & KEY_EXISTS(63,"AB ")  != 0)]  |
        E2EDKA1_OB      [? (KEY_EXISTS(0,"E2EDKA1") != 0 & KEY_EXISTS(63,"OB ")  != 0)]  |
        E2EDKA1_OMI     [? (KEY_EXISTS(0,"E2EDKA1") != 0 & KEY_EXISTS(63,"OMI") != 0)]  |
        E2EDKA1_OSF     [? (KEY_EXISTS(0,"E2EDKA1") != 0 & KEY_EXISTS(63,"OSF") != 0)]  |
        E2EDKA1_OPB     [? (KEY_EXISTS(0,"E2EDKA1") != 0 & KEY_EXISTS(63,"OPB") != 0)]  |
        E2EDKA1_OSU     [? (KEY_EXISTS(0,"E2EDKA1") != 0 & KEY_EXISTS(63,"OSU") != 0)]  |
        E2EDKA1         [? (KEY_EXISTS(0,"E2EDKA1") != 0
                           & KEY_EXISTS(63,"LF")     = 0
                           & KEY_EXISTS(63,"AG")     = 0
                           & KEY_EXISTS(63,"AP")     = 0
                           & KEY_EXISTS(63,"WE")     = 0

                           & KEY_EXISTS(63,"AB")     = 0
